# Поднять временный под, для изменения прав доступа

cat <<'EOF' | kubectl apply -f -
apiVersion: v1
kind: Pod
metadata:
  name: temp-files-export
  namespace: <TARGET_NS>
spec:
  restartPolicy: Never
  securityContext:
    runAsUser: 0
    fsGroup: 1000
  containers:
    - name: shell
      image: alpine:3.24.1
      command: ["sleep", "3600"]
      volumeMounts:
        - name: data
          mountPath: /data
  volumes:
    - name: data
      persistentVolumeClaim:
        claimName: <TARGET_CLAIM_NAME>
EOF

`kubectl -n <TARGET_NS> get pod temp-files-export -w   # дождись Running`
